\text { Caveman } 3\\
\begin{array}{l}
\int \frac{d x}{x \sqrt{1+x^{8}}} \\
\int \frac{d x}{x \sqrt{x^{8}\left(\frac{1}{x^{8}}+1\right)}}=\int \frac{d x}{x \cdot x^{4} \sqrt{x^{-8}+1}} \\
=\int \frac{d x}{x^{5} \sqrt{x^{-8}+1}}=\int \frac{x^{-5} d x}{\sqrt{x^{-8}+1}} \\
\begin{aligned}
u & =x^{-4} \rightarrow u^{2}=x^{-8} \\
d u & =-4 x^{-5} d x \\
-\frac{1}{4} d u & =x^{-5} d x
\end{aligned} \\
u=\tan \theta \\
d u=\sec ^{2} \theta d \theta
\end{array}\\
\begin{array}{l}
\text { I } \\
-\frac{1}{4} \int \frac{\sec ^{2} \theta}{\sqrt{\tan ^{2} \theta+1}} d \theta=-\frac{1}{4} \int \frac{\sec ^{2} \theta}{\sqrt{\sec ^{2} \theta}} d \theta \\
=-\frac{1}{4} \int \frac{\sec ^{2} \theta}{\sec \theta} d \theta=-\frac{1}{4} \int \sec \theta d \theta \\
=-\frac{1}{4} \ln |\sec \theta+\tan \theta|+C \\
=-\frac{1}{4} \ln \left|\sqrt{x^{-8}+1}+x^{-4}\right|+C \\
\sec \theta=\sqrt{u^{2}+1} \\
=\sqrt{x^{-8}+1} \\
\tan \theta=u \\
=x^{-4}
\end{array}
\end{array}

\text { Caveman Method } 2\\
\begin{array}{l}
\int \frac{x^{2}+1}{x^{4}+1} d x \\
\int \frac{\frac{x^{2}+1}{x^{2}}}{\frac{x^{4+1}}{x^{2}}} d x=\int \frac{1+\frac{1}{x^{2}}}{x^{2}+\frac{1}{x^{2}}} d x \\
\left(\begin{array}{l}
d u=1+\frac{1}{x^{2}} d x=1+x^{-2} d x \\
u=x-x^{-1} \\
\left(x-\frac{1}{x}\right)^{2}=x^{2}-2+\frac{1}{x^{2}}
\end{array}\right) \\
\int \frac{1+\frac{1}{x^{2}}}{x^{2}-2+\frac{1}{x^{2}}+2} d x \\
=\int \frac{1+\frac{1}{x^{2}}}{\left(x-\frac{1}{x}\right)^{2}}+2 d x \\
=\int \frac{1}{u^{2}+2} d u=\frac{1}{\sqrt{2}} \arctan \frac{u}{\sqrt{2}}+C \\
=\frac{1}{\sqrt{2}} \arctan \frac{x-\frac{1}{x}}{\sqrt{2}}+C
\end{array}
\end{array}
